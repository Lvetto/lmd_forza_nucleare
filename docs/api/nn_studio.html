 68    def setup_NN_basis(self):
 69        
 70        """
 71        Construct the NN basis states in the JLS coupling scheme, with isospin projection tz, and parity pi=(-1)^L. We include all states with jmin <= J <= jmax and tzmin <= tz <= tzmax, and we only include states that satisfy the Pauli principle, i.e., (L+S+T) must be odd.
 72
 73        Returns:
 74            list of dicts: A list of dictionaries representing the NN basis states.
 75        """
 76
 77        basis = []
 78        for tz in range(self.tzmin,self.tzmax+1,1):
 79            for J in range(self.jmin,self.jmax+1,1):
 80                for S in range(0,2,1):
 81                    for L in range(abs(J-S),J+S+1,1):
 82                        for T in range(abs(tz),2,1):
 83                            if ((L+S+T)%2 != 0):
 84                                basis_state = {}
 85                                basis_state['tz'] = tz
 86                                basis_state['l']  = L
 87                                basis_state['pi'] = (-1)**L
 88                                basis_state['s']  = S
 89                                basis_state['j']  = J
 90                                basis_state['t']  = T
 91                                basis.append(basis_state)
 92        return basis

 98
 99        Returns:
100            list of lists of dicts: A list of lists of dictionaries representing the NN channels.
101        """

103        from itertools import groupby
104        from operator import itemgetter
105

173    def gauss_legendre_line_mesh(self,a,b):
174
175        """
176        Construct a Gauss-Legendre mesh for the momentum integration. The mesh points and weights
177
178        Args:
179            a (float): The lower bound of the interval for the momentum integration.
180            b (float): The upper bound of the interval for the momentum integration.
181
182        Returns:
183            tuple: A tuple containing the mesh points and weights.
184        """
185
186        x, w = np.polynomial.legendre.leggauss(self.Np)
187        # Translate x values from the interval [-1, 1] to [a, b]
188        t = 0.5*(x + 1)*(b - a) + a
189        u = w * 0.5*(b - a)
190
191        return t,u

193    def gauss_legendre_inf_mesh(self):
194
195        """
196        Construct a Gauss-Legendre mesh for the momentum integration on the interval [0, inf). The mesh points and weights are given by the Gauss-Legendre quadrature on the interval [-1, 1], but translated to the interval [0, inf) using the transformation t = scale * tan(pi/4 * (x + 1)), where x are the Gauss-Legendre points. The weights are also transformed accordingly.
197
198        Returns:
199            tuple: A tuple containing the mesh points and weights.
200        """
201
202        scale=100.0
203        
204        x, w = np.polynomial.legendre.leggauss(self.Np)
205        
206        # Translate x values from the interval [-1, 1] to [0, inf)
207        pi_over_4 = np.pi/4.0
208        
209        t = scale*np.tan(pi_over_4*(x+1.0))
210        u = scale*pi_over_4/np.cos(pi_over_4*(x+1.0))**2*w
211        
212        return t,u

257    def lab2rel(self,Tlab,tz):
258
259        """
260        Convert the laboratory kinetic energy Tlab to the relative momentum ko and the reduced mass mu for the given isospin projection tz. The formulas for the conversion depend on the value of tz, which determines the type of scattering (pp, np, or nn).
261
262        Args:
263            Tlab (float): The laboratory kinetic energy in MeV.
264            tz (int): The isospin projection, where tz=-1 corresponds to pp scattering, tz=0 corresponds to np scattering, and tz=+1 corresponds to nn scattering.
265        
266        Returns:
267            tuple: A tuple containing the relative momentum ko and the reduced mass mu.
268        """
269    
270        if tz == -1:
271            mu = const.Mp/2
272            ko2 = 2*const.Mp*Tlab
273        elif tz ==  0:
274            mu = const.Mp*const.Mn/(const.Mp+const.Mn)
275            ko2 = const.Mp**2*Tlab*(Tlab+2*const.Mn)/((const.Mp+const.Mn)**2 + 2*Tlab*const.Mp)
276        elif tz == +1:
277            mu = const.mN/2
278            ko2 = 2*const.Mp*Tlab
279        else:
280            exit('unknown isospin projection')
281
282        if ko2<0:
283            ko = np.complex(0,np.sqrt(np.abs(ko2)))
284        else:    
285            ko = np.sqrt(ko2)
286
287        return ko,mu

291    def map_to_coup_idx(ll,l,s,j):
292
293        """
294        Map the quantum numbers of the partial-wave coupling to the index of the potential matrix element. The mapping is based on the coupling scheme and the order of the potential matrix elements in the code.
295        The mapping is as follows:
296
297        l=ll, s=1, l<j: idx=3 (--)
298
299        l=ll, s=1, l>j: idx=2 (++)
300
301        l=ll, s=1, l=j: idx=1 (uncoupled)
302
303        l=ll, s=0, l=j: idx=0 (uncoupled)
304
305        l!=ll, l<j: idx=5 (-+)
306
307        l!=ll, l>j: idx=4 (+-)
308
309        Args:
310            ll (int): The orbital angular momentum of the ket state.
311            l (int): The orbital angular momentum of the bra state.
312            s (int): The total spin of the two nucleons.
313            j (int): The total angular momentum of the two nucleons.
314
315        Returns:
316            tuple: A tuple containing a boolean indicating if the coupling is valid and the index of the potential matrix element.
317        """
318
319        if l == ll:
320        
321            if l<j:
322                # --
323                coup = True
324                idx  = 3
325            elif l>j:
326                # ++
327                coup = True
328                idx  = 2
329            else:
330                if s==1:
331                    coup = False
332                    idx  = 1
333                else:
334                    coup = False
335                    idx  = 0
336        else:
337            if l<j:
338                # -+
339                coup = True
340                idx  = 5
341            else:
342                # +-
343                coup = True
344                idx  = 4
345
346        return coup,idx

348    def Vmtx(self,this_mesh,ll,l,s,j,t,tz):
349
350        """
351        Construct the potential matrix for the given quantum numbers and the momentum mesh.
352        The potential matrix is constructed by evaluating the potential function for each pair of momentum points in the mesh, and for the given quantum numbers of the partial-wave coupling.
353
354        Args:
355            this_mesh (np.ndarray): The momentum mesh for which to construct the potential matrix.
356            ll (int): The orbital angular momentum of the ket state.
357            l (int): The orbital angular momentum of the bra state.
358            s (int): The total spin of the two nucleons.
359            j (int): The total angular momentum of the two nucleons.
360            t (int): The total isospin of the two nucleons.
361            tz (int): The isospin projection of the two nucleons.
362
363        Returns:
364            np.ndarray: The potential matrix for the given quantum numbers and momentum mesh.
365        """
366    
367        coup,idx = self.map_to_coup_idx(ll,l,s,j)
368        mtx = np.zeros((len(this_mesh), len(this_mesh)))
369        for pidx, p in enumerate(this_mesh):
370            for ppidx, pp in enumerate(this_mesh):
371                mtx[ppidx][pidx] = self.V.potential(pp,p,coup,s,j,t,tz,self.lecs)[idx]
372        return np.array(mtx)

415    def setup_G0_vector(self,ko,mu):
416
417        """
418        Construct the G0 vector for the given relative momentum ko and reduced mass mu.
419        The G0 vector is constructed by evaluating the free Green's function for each momentum point in the mesh, and for the on-shell momentum ko.
420
421        Args:
422            ko (float): The relative momentum.
423            mu (float): The reduced mass.
424
425        Returns:
426            np.ndarray: The G0 vector for the given relative momentum ko and reduced mass mu.
427        """
428            
429        G = np.zeros((2*self.Np+2), dtype=complex)
430
431        # note that we index from zero, and the N+1 point is at self.Np
432        G[0:self.Np] = self.wmesh*self.pmesh**2/(ko**2 - self.pmesh**2)		# Gaussian integral
433
434        #print('   G0 pole subtraction')
435        G[self.Np]  = -np.sum( self.wmesh/(ko**2 - self.pmesh**2 ) )*ko**2 	# 'Principal value'
436        G[self.Np] -= 1j*ko * (np.pi/2)
437
438        #python vec[0:n] is the first n elements, i.e., 0,1,2,3,...,n-1
439        G[self.Np+1:2*self.Np+2] = G[0:self.Np+1]
440        return G*2*mu

442    def setup_GV_kernel(self,channel,Vmtx,ko,mu):
443
444        """
445        Construct the GV kernel for the given channel, potential matrix, relative momentum ko, and reduced mass mu.
446
447        Args:
448            channel (list): The channel for which to construct the GV kernel. The channel is a list of blocks, where each block is a dictionary containing the quantum numbers of the partial-wave coupling.
449            Vmtx (np.ndarray): The potential matrix for the given channel and momentum mesh.
450            ko (float): The relative momentum.
451            mu (float): The reduced mass.
452
453        Returns:
454            np.ndarray: The GV kernel for the given channel, potential matrix, relative momentum ko, and reduced mass mu.
455        """
456    
457        Np = len(self.pmesh)
458        nof_blocks = len(channel)
459        Np_chn = int(np.sqrt(nof_blocks)*(self.Np+1))
460        # Go-vector dim(u) = 2*len(p)+2
461        G0 = self.setup_G0_vector(ko,mu)
462        
463        g = np.copy(G0[0:Np_chn])
464        GV = np.zeros((len(g),len(g)),dtype=complex)
465        
466        for g_idx, g_elem in enumerate(g):
467            GV[g_idx,:] = g_elem * Vmtx[g_idx,:]
468            
469        return GV

500    def solve_lippmann_schwinger(self,channel,Vmtx,ko,mu):
501
502        """
503        Solve the Lippmann-Schwinger equation for the given channel, potential matrix, relative momentum ko, and reduced mass mu.
504        The Lippmann-Schwinger equation is solved by matrix inversion, using the GV and VG kernels.
505
506        Args:
507            channel (list): The channel for which to solve the Lippmann-Schwinger equation. The channel is a list of blocks, where each block is a dictionary containing the quantum numbers of the partial-wave coupling.
508            Vmtx (np.ndarray): The potential matrix for the given channel and momentum mesh.
509            ko (float): The relative momentum.
510            mu (float): The reduced mass.
511
512        Returns:
513            np.ndarray: The solution to the Lippmann-Schwinger equation for the given channel, potential matrix, relative momentum ko, and reduced mass mu.
514        """
515
516        # matrix inversion:
517        # T = V + VGT
518        # (1-VG)T = V
519        # T = (1-VG)^{-1}V
520        
521        VG = self.setup_VG_kernel(channel,Vmtx,ko,mu)
522        VG = np.eye(VG.shape[0]) - VG
523        # golden rule of linear algebra: avoid matrix inversion if you can
524        #T = np.matmul(np.linalg.inv(VG),Vmtx)
525        T = np.linalg.solve(VG,Vmtx)
526
527        return T

536

538            ko (float): The relative momentum.
539            mu (float): The reduced mass.
540            on_shell_T (list): A list of on-shell T-matrix elements, where the first element is T11, the second element is T12, and the third element is T22 for coupled channels, and the first element is T11 for uncoupled channels.
541
542        Returns:
543            np.ndarray: The computed phase shifts for the given relative momentum ko, reduced mass mu, and on-shell T-matrix elements.
544        """
545
546        rad2deg = 180.0/np.pi
547        
548        fac  = np.pi*mu*ko
549        
550        if len(on_shell_T) == 3:
551            
552            T11 = on_shell_T[0]
553            T12 = on_shell_T[1]
554            T22 = on_shell_T[2]
555        
556            # Blatt-Biedenharn (BB) convention
557            twoEpsilonJ_BB = np.arctan(2*T12/(T11-T22))	# mixing parameter
558            delta_plus_BB  = -0.5*1j*np.log(1 - 1j*fac*(T11+T22) + 1j*fac*(2*T12)/np.sin(twoEpsilonJ_BB))
559            delta_minus_BB = -0.5*1j*np.log(1 - 1j*fac*(T11+T22) - 1j*fac*(2*T12)/np.sin(twoEpsilonJ_BB))
560

563            #twoEpsilonJ = np.arcsin(np.sin(twoEpsilonJ_BB)*np.sin(delta_minus_BB - delta_plus_BB))      # mixing parameter
564            #delta_minus = 0.5*(delta_plus_BB + delta_minus_BB + np.arcsin(np.tan(twoEpsilonJ)/np.tan(twoEpsilonJ_BB)))
565            #delta_plus  = 0.5*(delta_plus_BB + delta_minus_BB - np.arcsin(np.tan(twoEpsilonJ)/np.tan(twoEpsilonJ_BB)))
566            #epsilon     = 0.5*twoEpsilonJ
567
568            # numerially stable conversion
569            cos2e = np.cos(twoEpsilonJ_BB/2)*np.cos(twoEpsilonJ_BB/2)
570            cos_2dp = np.cos(2*delta_plus_BB)
571            cos_2dm = np.cos(2*delta_minus_BB)
572            sin_2dp = np.sin(2*delta_plus_BB)
573            sin_2dm = np.sin(2*delta_minus_BB)
574            
575            aR = np.real(cos2e*cos_2dm + (1-cos2e)*cos_2dp)
576            aI = np.real(cos2e*sin_2dm + (1-cos2e)*sin_2dp)
577            delta_minus = 0.5*np.arctan2(aI,aR)
578
579            aR = np.real(cos2e*cos_2dp + (1-cos2e)*cos_2dm)
580            aI = np.real(cos2e*sin_2dp + (1-cos2e)*sin_2dm)
581            delta_plus = 0.5*np.arctan2(aI,aR)
582
583            tmp = 0.5*np.sin(twoEpsilonJ_BB)
584            aR = tmp*(cos_2dm - cos_2dp)
585            aI = tmp*(sin_2dm - sin_2dp)
586            tmp = delta_plus + delta_minus
587            epsilon = 0.5*np.arcsin(aI*np.cos(tmp) - aR*np.sin(tmp)) 
588            
589            if ko <150:
590                if delta_minus*rad2deg<0:
591                    delta_minus += np.pi
592                    epsilon *= -1.0
593            return [np.real(delta_minus*rad2deg), np.real(delta_plus*rad2deg), np.real(epsilon*rad2deg)]
594        
595        else:
596            # uncoupled
597            T = on_shell_T[0]
598            Z = 1-fac*2j*T
599            # S=exp(2i*delta)
600            delta = (-0.5*1j)*np.log(Z)
601
602            return np.real(delta*rad2deg)

680    def model_order_reduction(self,channel,Tlab,directions,training_points,verbose=False):
681
682        """
683        Perform model order reduction for the given channel, laboratory kinetic energy Tlab, directions of variation for the low-energy constants (LECs), and training points for the LECs.
684        The model order reduction is performed by splitting the potential matrix into a constant part and split parts corresponding to the variation of the LECs in the specified directions, and then constructing the emulator by computing the relevant kernels and solving the Lippmann-Schwinger equation for the training points.
685
686        Args:
687            channel (list): The channel for which to perform model order reduction. The channel is a list of blocks, where each block is a dictionary containing the quantum numbers of the partial-wave coupling.
688            Tlab (float): The laboratory kinetic energy for which to perform model order reduction.
689            directions (list): A list of directions of variation for the low-energy constants (LECs), where each direction is specified by the name of the LEC to vary.
690            training_points (list): A list of training points for the LECs, where each training point is a list of LEC values corresponding to the specified directions.
691            verbose (bool): Whether to print verbose output.
692
693        Returns:
694            function: The reduced model function.
695        """
696
697        if len(channel) > 1:
698            exit('model order reduction: limited to one channel at the time')
699
700        if len(directions) != len(training_points[0]):
701            exit('model order reduction: need training in every direction')
702            
703        nof_blocks = len(channel[0])
704        Np_chn = int(np.sqrt(nof_blocks)*(self.Np+1))
705        Np = Np_chn-1
706        if nof_blocks == 4:
707            Np = int((Np_chn-2)/2)
708                    
709        if verbose:
710            print(f'MOR:')
711            print(f'channel          = {channel}')
712            print(f'nof_blocks       = {nof_blocks}')
713            print(f'Np_chn           = {Np_chn}')
714            print(f'Tlab             = {Tlab}')
715            print(f'directions       = {directions}')
716            print(f'#training points = {len(training_points)}')
717            print(f'{training_points}')
718
719        # we use nof_blocks to identify coupled (=4) and uncoupled (=1) channels
720                
721        ko,mu= self.lab2rel(Tlab,channel[0][0]['tz'])
722        
723        # we construct a diagonal matrix from the vector G0
724        G = np.diag(self.setup_G0_vector(ko,mu)[0:Np_chn])
725        
726        # set relevant lecs to zero to extract constant part
727        for lec, value in self.lecs.items():
728            if lec in directions:
729                self.lecs[lec] = 0
730                
731        V_split = []
732        V_split.append(self.setup_Vmtx(channel[0],ko)[0]) # get only V, not the list of submatrices
733
734        # set lecs in each direction to 1.0 and get the split potential term
735        for lec, value in self.lecs.items():
736            if lec in directions:
737                self.lecs[lec] = 1.0
738                V_direction = self.setup_Vmtx(channel[0],ko)[0] # get only V, not the list of submatrices
739                V_split.append(V_direction - V_split[0])
740                self.lecs[lec] = 0.0
741                    
742        # we need GV, VG for each split V term
743        GV_split = []
744        VG_split = []
745        GVG_split = []
746        Vi_split_T11 = [] # the on-shell Vi-split terms
747        Vi_split_T12 = [] # the on-shell Vi-split terms
748        Vi_split_T22 = [] # the on-shell Vi-split terms
749        for this_V in V_split:
750            GV = self.setup_GV_kernel(channel[0],this_V,ko,mu)
751            VG = self.setup_VG_kernel(channel[0],this_V,ko,mu)
752            Vi_split_T11.append(this_V[Np,Np])
753            if (nof_blocks == 4):
754                Vi_split_T12.append(this_V[2*Np+1,Np])
755                Vi_split_T22.append(this_V[2*Np+1,2*Np+1])
756            GV_split.append(GV)
757            VG_split.append(VG)
758            GVG_split.append(GV@G)
759                
760        # loop over the training lec values and setup the relevant lec vector
761        # to assemble the potential for the split terms
762        Ti = []
763        for training_point in training_points:
764            lec_training_vector = []
765            lec_training_vector.append(1.0)
766            for lec, value in self.lecs.items():
767                if lec in directions:
768                    # training points and directions are identically ordered wrt lec names
769                    lec_training_vector.append(training_point[directions.index(lec)])
770
771            V = self.get_Vmtx_from_split(V_split,lec_training_vector)
772            this_T = self.solve_lippmann_schwinger(channel[0],V,ko,mu)
773            Ti.append(self.solve_lippmann_schwinger(channel[0],V,ko,mu))
774
775        # construct the emulator
776        mi_split_T11  = np.zeros((len(training_points)), dtype=object)
777        Mij_split_T11 = np.zeros((len(training_points),len(training_points)), dtype=object)
778        Mij_const_T11 = np.zeros((len(training_points),len(training_points)), dtype=object)
779
780        if (nof_blocks == 4):
781            mi_split_T12  = np.zeros((len(training_points)), dtype=object)
782            mi_split_T22  = np.zeros((len(training_points)), dtype=object)
783            Mij_split_T12 = np.zeros((len(training_points),len(training_points)), dtype=object)
784            Mij_split_T22 = np.zeros((len(training_points),len(training_points)), dtype=object)

786            Mij_const_T22 = np.zeros((len(training_points),len(training_points)), dtype=object)
787            
788        for r,training_point in enumerate(training_points):
789            mi_split_T11[r] = []
790            if (nof_blocks == 4):
791                mi_split_T12[r] = []
792                mi_split_T22[r] = []
793            for c,training_point in enumerate(training_points):
794                Mij_split_T11[r,c] = []
795                Mij_const_T11[r,c] = []
796                if (nof_blocks == 4):
797                    Mij_split_T12[r,c] = []
798                    Mij_const_T12[r,c] = []
799                    Mij_split_T22[r,c] = []
800                    Mij_const_T22[r,c] = []
801                    
802        for split_idx, _ in enumerate(GV_split):
803            for r,training_point in enumerate(training_points):
804                this_mi_TGV = np.copy(Ti[r]@GV_split[split_idx])
805                this_mi_VGT = np.copy(VG_split[split_idx]@Ti[r])
806                mi_split_T11[r].append((this_mi_TGV + this_mi_VGT)[Np,Np])
807                if (nof_blocks == 4):
808                    mi_split_T12[r].append((this_mi_TGV + this_mi_VGT)[2*Np+1,Np])
809                    mi_split_T22[r].append((this_mi_TGV + this_mi_VGT)[2*Np+1,2*Np+1])
810                for c,training_point in enumerate(training_points):
811                    this_TGVGT_rc = np.copy(-Ti[r]@GVG_split[split_idx]@Ti[c])
812                    this_TGVGT_cr = np.copy(-Ti[c]@GVG_split[split_idx]@Ti[r])
813                    Mij_split_T11[r,c].append((this_TGVGT_rc + this_TGVGT_cr)[Np,Np])
814                    Mij_const_T11[r,c] = (Ti[c]@G@Ti[r] + Ti[r]@G@Ti[c])[Np,Np]
815                    if (nof_blocks == 4):
816                        Mij_split_T12[r,c].append((this_TGVGT_rc + this_TGVGT_cr)[2*Np+1,Np])

818                        Mij_split_T22[r,c].append((this_TGVGT_rc + this_TGVGT_cr)[2*Np+1,2*Np+1])
819                        Mij_const_T22[r,c] = (Ti[c]@G@Ti[r] + Ti[r]@G@Ti[c])[2*Np+1,2*Np+1]
820                       
821        def emulator(emulator_lecs,verbose=False):
822
823            ntp = len(training_points)
824            this_ko   = ko
825            this_mu   = mu
826            this_Tlab = Tlab
827            this_channel = channel
828            this_nof_blocks = nof_blocks
829            
830            if verbose:
831                print(f'emulating for lecs = {emulator_lecs}')
832            
833            # construct m and M
834            nof_split_terms = len(mi_split_T11[0])
835            nof_emulator_lecs = len(emulator_lecs)
836            
837            m_i_T11 = np.zeros((ntp), dtype=complex)
838            M_ij_T11 = np.zeros((ntp,ntp), dtype=complex)
839            V_T11 = 0
840            if this_nof_blocks==4:
841                m_i_T12 = np.zeros((ntp), dtype=complex)
842                M_ij_T12 = np.zeros((ntp,ntp), dtype=complex)
843                V_T12 = 0
844                m_i_T22 = np.zeros((ntp), dtype=complex)
845                M_ij_T22 = np.zeros((ntp,ntp), dtype=complex)
846                V_T22 = 0
847                
848            assert nof_split_terms == nof_emulator_lecs , f"{nof_split_terms} /= {nof_emulator_lecs}"
849            for r in range(0,ntp):
850                for idx,lec in enumerate(emulator_lecs):
851                    m_i_T11[r]  += lec*mi_split_T11[r][idx]
852                    if this_nof_blocks == 4:
853                        m_i_T12[r]  += lec*mi_split_T12[r][idx]
854                        m_i_T22[r]  += lec*mi_split_T22[r][idx]
855                for c in range(0,ntp):
856                    for idx,lec in enumerate(emulator_lecs):
857                        M_ij_T11[r,c] += lec*Mij_split_T11[r,c][idx]
858                        if this_nof_blocks == 4:
859                            M_ij_T12[r,c] += lec*Mij_split_T12[r,c][idx]
860                            M_ij_T22[r,c] += lec*Mij_split_T22[r,c][idx]
861                            
862                    M_ij_T11[r,c] += Mij_const_T11[r,c]
863                    if this_nof_blocks == 4:
864                        M_ij_T12[r,c] += Mij_const_T12[r,c]
865                        M_ij_T22[r,c] += Mij_const_T22[r,c]
866                        
867            for idx,lec in enumerate(emulator_lecs):
868                V_T11 += lec*Vi_split_T11[idx]
869                if this_nof_blocks == 4:
870                    V_T12 += lec*Vi_split_T12[idx]
871                    V_T22 += lec*Vi_split_T22[idx]
872                    
873            T11 = V_T11 + 0.5*m_i_T11.T@np.linalg.inv(M_ij_T11)@m_i_T11

877
878            if this_nof_blocks == 4:
879                return [T11,T12,T22],this_Tlab,this_ko,this_mu
880            return [T11],this_Tlab,this_ko,this_mu

882        return emulator

263        Args:
264            Tlab (float): The laboratory kinetic energy in MeV.
265            tz (int): The isospin projection, where tz=-1 corresponds to pp scattering, tz=0 corresponds to np scattering, and tz=+1 corresponds to nn scattering.
266        
267        Returns:
268            tuple: A tuple containing the relative momentum ko and the reduced mass mu.
269        """
270    
271        if tz == -1:
272            mu = const.Mp/2
273            ko2 = 2*const.Mp*Tlab
274        elif tz ==  0:
275            mu = const.Mp*const.Mn/(const.Mp+const.Mn)
276            ko2 = const.Mp**2*Tlab*(Tlab+2*const.Mn)/((const.Mp+const.Mn)**2 + 2*Tlab*const.Mp)
277        elif tz == +1:
278            mu = const.mN/2
279            ko2 = 2*const.Mp*Tlab
280        else:
281            exit('unknown isospin projection')
282
283        if ko2<0:
284            ko = np.complex(0,np.sqrt(np.abs(ko2)))
285        else:    
286            ko = np.sqrt(ko2)
287
288        return ko,mu

290    @staticmethod
291    #a static method is bound to a class rather than the objects for that class
292    def map_to_coup_idx(ll,l,s,j):
293
294        """
295        Map the quantum numbers of the partial-wave coupling to the index of the potential matrix element. The mapping is based on the coupling scheme and the order of the potential matrix elements in the code.
296        The mapping is as follows:
297
298        l=ll, s=1, l<j: idx=3 (--)
299
300        l=ll, s=1, l>j: idx=2 (++)
301
302        l=ll, s=1, l=j: idx=1 (uncoupled)
303
304        l=ll, s=0, l=j: idx=0 (uncoupled)
305
306        l!=ll, l<j: idx=5 (-+)
307
308        l!=ll, l>j: idx=4 (+-)
309
310        Args:
311            ll (int): The orbital angular momentum of the ket state.
312            l (int): The orbital angular momentum of the bra state.
313            s (int): The total spin of the two nucleons.
314            j (int): The total angular momentum of the two nucleons.
315
316        Returns:
317            tuple: A tuple containing a boolean indicating if the coupling is valid and the index of the potential matrix element.
318        """
319
320        if l == ll:
321        
322            if l<j:
323                # --
324                coup = True
325                idx  = 3
326            elif l>j:
327                # ++
328                coup = True
329                idx  = 2
330            else:
331                if s==1:
332                    coup = False
333                    idx  = 1
334                else:
335                    coup = False
336                    idx  = 0
337        else:
338            if l<j:
339                # -+
340                coup = True
341                idx  = 5
342            else:
343                # +-
344                coup = True
345                idx  = 4
346
347        return coup,idx

349    def Vmtx(self,this_mesh,ll,l,s,j,t,tz):
350
351        """
352        Construct the potential matrix for the given quantum numbers and the momentum mesh.
353        The potential matrix is constructed by evaluating the potential function for each pair of momentum points in the mesh, and for the given quantum numbers of the partial-wave coupling.
354
355        Args:
356            this_mesh (np.ndarray): The momentum mesh for which to construct the potential matrix.
357            ll (int): The orbital angular momentum of the ket state.
358            l (int): The orbital angular momentum of the bra state.
359            s (int): The total spin of the two nucleons.
360            j (int): The total angular momentum of the two nucleons.
361            t (int): The total isospin of the two nucleons.
362            tz (int): The isospin projection of the two nucleons.
363
364        Returns:
365            np.ndarray: The potential matrix for the given quantum numbers and momentum mesh.
366        """
367    
368        coup,idx = self.map_to_coup_idx(ll,l,s,j)
369        mtx = np.zeros((len(this_mesh), len(this_mesh)))
370        for pidx, p in enumerate(this_mesh):
371            for ppidx, pp in enumerate(this_mesh):
372                mtx[ppidx][pidx] = self.V.potential(pp,p,coup,s,j,t,tz,self.lecs)[idx]
373        return np.array(mtx)

416    def setup_G0_vector(self,ko,mu):
417
418        """
419        Construct the G0 vector for the given relative momentum ko and reduced mass mu.
420        The G0 vector is constructed by evaluating the free Green's function for each momentum point in the mesh, and for the on-shell momentum ko.
421
422        Args:
423            ko (float): The relative momentum.
424            mu (float): The reduced mass.
425
426        Returns:
427            np.ndarray: The G0 vector for the given relative momentum ko and reduced mass mu.
428        """
429            
430        G = np.zeros((2*self.Np+2), dtype=complex)
431
432        # note that we index from zero, and the N+1 point is at self.Np
433        G[0:self.Np] = self.wmesh*self.pmesh**2/(ko**2 - self.pmesh**2)		# Gaussian integral
434
435        #print('   G0 pole subtraction')
436        G[self.Np]  = -np.sum( self.wmesh/(ko**2 - self.pmesh**2 ) )*ko**2 	# 'Principal value'
437        G[self.Np] -= 1j*ko * (np.pi/2)
438
439        #python vec[0:n] is the first n elements, i.e., 0,1,2,3,...,n-1
440        G[self.Np+1:2*self.Np+2] = G[0:self.Np+1]
441        return G*2*mu

443    def setup_GV_kernel(self,channel,Vmtx,ko,mu):
444
445        """
446        Construct the GV kernel for the given channel, potential matrix, relative momentum ko, and reduced mass mu.
447
448        Args:
449            channel (list): The channel for which to construct the GV kernel. The channel is a list of blocks, where each block is a dictionary containing the quantum numbers of the partial-wave coupling.
450            Vmtx (np.ndarray): The potential matrix for the given channel and momentum mesh.
451            ko (float): The relative momentum.
452            mu (float): The reduced mass.
453
454        Returns:
455            np.ndarray: The GV kernel for the given channel, potential matrix, relative momentum ko, and reduced mass mu.
456        """
457    
458        Np = len(self.pmesh)
459        nof_blocks = len(channel)
460        Np_chn = int(np.sqrt(nof_blocks)*(self.Np+1))
461        # Go-vector dim(u) = 2*len(p)+2
462        G0 = self.setup_G0_vector(ko,mu)
463        
464        g = np.copy(G0[0:Np_chn])
465        GV = np.zeros((len(g),len(g)),dtype=complex)
466        
467        for g_idx, g_elem in enumerate(g):
468            GV[g_idx,:] = g_elem * Vmtx[g_idx,:]
469            
470        return GV

473
474        """
475        Construct the VG kernel for the given channel, potential matrix, relative momentum ko, and reduced mass mu.
476
477        Args:
478            channel (list): The channel for which to construct the VG kernel. The channel is a list of blocks, where each block is a dictionary containing the quantum numbers of the partial-wave coupling.
479            Vmtx (np.ndarray): The potential matrix for the given channel and momentum mesh.
480            ko (float): The relative momentum.
481            mu (float): The reduced mass.
482
483        Returns:
484            np.ndarray: The VG kernel for the given channel, potential matrix, relative momentum ko, and reduced mass mu.
485        """
486
487        Np = len(self.pmesh)
488        nof_blocks = len(channel)
489        Np_chn = int(np.sqrt(nof_blocks)*(self.Np+1))
490        
491        # Go-vector dim(u) = 2*len(p)+2
492        G0 = self.setup_G0_vector(ko,mu)

501    def solve_lippmann_schwinger(self,channel,Vmtx,ko,mu):
502
503        """
504        Solve the Lippmann-Schwinger equation for the given channel, potential matrix, relative momentum ko, and reduced mass mu.
505        The Lippmann-Schwinger equation is solved by matrix inversion, using the GV and VG kernels.
506
507        Args:
508            channel (list): The channel for which to solve the Lippmann-Schwinger equation. The channel is a list of blocks, where each block is a dictionary containing the quantum numbers of the partial-wave coupling.
509            Vmtx (np.ndarray): The potential matrix for the given channel and momentum mesh.
510            ko (float): The relative momentum.
511            mu (float): The reduced mass.
512
513        Returns:
514            np.ndarray: The solution to the Lippmann-Schwinger equation for the given channel, potential matrix, relative momentum ko, and reduced mass mu.
515        """
516
517        # matrix inversion:
518        # T = V + VGT
519        # (1-VG)T = V
520        # T = (1-VG)^{-1}V
521        
522        VG = self.setup_VG_kernel(channel,Vmtx,ko,mu)
523        VG = np.eye(VG.shape[0]) - VG
524        # golden rule of linear algebra: avoid matrix inversion if you can
525        #T = np.matmul(np.linalg.inv(VG),Vmtx)
526        T = np.linalg.solve(VG,Vmtx)
527
528        return T

530    @staticmethod
531    #a static method is bound to a class rather than the objects for that class
532    def compute_phase_shifts(ko,mu,on_shell_T):
533
534        """
535        Compute the phase shifts for the given relative momentum ko, reduced mass mu, and on-shell T-matrix elements.
536        The phase shifts are computed using the formulas for the Blatt-Biedenharn convention for coupled channels, and the standard formula for uncoupled channels.
537
538        Args:
539            ko (float): The relative momentum.
540            mu (float): The reduced mass.
541            on_shell_T (list): A list of on-shell T-matrix elements, where the first element is T11, the second element is T12, and the third element is T22 for coupled channels, and the first element is T11 for uncoupled channels.
542
543        Returns:
544            np.ndarray: The computed phase shifts for the given relative momentum ko, reduced mass mu, and on-shell T-matrix elements.
545        """
546
547        rad2deg = 180.0/np.pi
548        
549        fac  = np.pi*mu*ko
550        
551        if len(on_shell_T) == 3:
552            
553            T11 = on_shell_T[0]
554            T12 = on_shell_T[1]
555            T22 = on_shell_T[2]
556        
557            # Blatt-Biedenharn (BB) convention
558            twoEpsilonJ_BB = np.arctan(2*T12/(T11-T22))	# mixing parameter
559            delta_plus_BB  = -0.5*1j*np.log(1 - 1j*fac*(T11+T22) + 1j*fac*(2*T12)/np.sin(twoEpsilonJ_BB))
560            delta_minus_BB = -0.5*1j*np.log(1 - 1j*fac*(T11+T22) - 1j*fac*(2*T12)/np.sin(twoEpsilonJ_BB))
561
562            # this version has a numerical instability that I should fix.
563            # Stapp convention (bar-phase shifts) in terms of Blatt-Biedenharn convention
564            #twoEpsilonJ = np.arcsin(np.sin(twoEpsilonJ_BB)*np.sin(delta_minus_BB - delta_plus_BB))      # mixing parameter
565            #delta_minus = 0.5*(delta_plus_BB + delta_minus_BB + np.arcsin(np.tan(twoEpsilonJ)/np.tan(twoEpsilonJ_BB)))
566            #delta_plus  = 0.5*(delta_plus_BB + delta_minus_BB - np.arcsin(np.tan(twoEpsilonJ)/np.tan(twoEpsilonJ_BB)))
567            #epsilon     = 0.5*twoEpsilonJ
568
569            # numerially stable conversion
570            cos2e = np.cos(twoEpsilonJ_BB/2)*np.cos(twoEpsilonJ_BB/2)
571            cos_2dp = np.cos(2*delta_plus_BB)
572            cos_2dm = np.cos(2*delta_minus_BB)
573            sin_2dp = np.sin(2*delta_plus_BB)
574            sin_2dm = np.sin(2*delta_minus_BB)
575            
576            aR = np.real(cos2e*cos_2dm + (1-cos2e)*cos_2dp)
577            aI = np.real(cos2e*sin_2dm + (1-cos2e)*sin_2dp)
578            delta_minus = 0.5*np.arctan2(aI,aR)
579
580            aR = np.real(cos2e*cos_2dp + (1-cos2e)*cos_2dm)
581            aI = np.real(cos2e*sin_2dp + (1-cos2e)*sin_2dm)
582            delta_plus = 0.5*np.arctan2(aI,aR)
583
584            tmp = 0.5*np.sin(twoEpsilonJ_BB)
585            aR = tmp*(cos_2dm - cos_2dp)
586            aI = tmp*(sin_2dm - sin_2dp)
587            tmp = delta_plus + delta_minus
588            epsilon = 0.5*np.arcsin(aI*np.cos(tmp) - aR*np.sin(tmp)) 
589            
590            if ko <150:
591                if delta_minus*rad2deg<0:
592                    delta_minus += np.pi
593                    epsilon *= -1.0
594            return [np.real(delta_minus*rad2deg), np.real(delta_plus*rad2deg), np.real(epsilon*rad2deg)]
595        
596        else:
597            # uncoupled
598            T = on_shell_T[0]
599            Z = 1-fac*2j*T
600            # S=exp(2i*delta)
601            delta = (-0.5*1j)*np.log(Z)
602
603            return np.real(delta*rad2deg)

681    def model_order_reduction(self,channel,Tlab,directions,training_points,verbose=False):
682
683        """
684        Perform model order reduction for the given channel, laboratory kinetic energy Tlab, directions of variation for the low-energy constants (LECs), and training points for the LECs.
685        The model order reduction is performed by splitting the potential matrix into a constant part and split parts corresponding to the variation of the LECs in the specified directions, and then constructing the emulator by computing the relevant kernels and solving the Lippmann-Schwinger equation for the training points.
686
687        Args:
688            channel (list): The channel for which to perform model order reduction. The channel is a list of blocks, where each block is a dictionary containing the quantum numbers of the partial-wave coupling.
689            Tlab (float): The laboratory kinetic energy for which to perform model order reduction.
690            directions (list): A list of directions of variation for the low-energy constants (LECs), where each direction is specified by the name of the LEC to vary.
691            training_points (list): A list of training points for the LECs, where each training point is a list of LEC values corresponding to the specified directions.
692            verbose (bool): Whether to print verbose output.
693
694        Returns:
695            function: The reduced model function.
696        """
697
698        if len(channel) > 1:
699            exit('model order reduction: limited to one channel at the time')
700
701        if len(directions) != len(training_points[0]):
702            exit('model order reduction: need training in every direction')
703            
704        nof_blocks = len(channel[0])
705        Np_chn = int(np.sqrt(nof_blocks)*(self.Np+1))
706        Np = Np_chn-1
707        if nof_blocks == 4:
708            Np = int((Np_chn-2)/2)
709                    
710        if verbose:
711            print(f'MOR:')
712            print(f'channel          = {channel}')
713            print(f'nof_blocks       = {nof_blocks}')
714            print(f'Np_chn           = {Np_chn}')
715            print(f'Tlab             = {Tlab}')
716            print(f'directions       = {directions}')
717            print(f'#training points = {len(training_points)}')
718            print(f'{training_points}')
719
720        # we use nof_blocks to identify coupled (=4) and uncoupled (=1) channels
721                
722        ko,mu= self.lab2rel(Tlab,channel[0][0]['tz'])
723        
724        # we construct a diagonal matrix from the vector G0
725        G = np.diag(self.setup_G0_vector(ko,mu)[0:Np_chn])
726        
727        # set relevant lecs to zero to extract constant part
728        for lec, value in self.lecs.items():
729            if lec in directions:
730                self.lecs[lec] = 0
731                
732        V_split = []
733        V_split.append(self.setup_Vmtx(channel[0],ko)[0]) # get only V, not the list of submatrices
734
735        # set lecs in each direction to 1.0 and get the split potential term
736        for lec, value in self.lecs.items():
737            if lec in directions:
738                self.lecs[lec] = 1.0
739                V_direction = self.setup_Vmtx(channel[0],ko)[0] # get only V, not the list of submatrices
740                V_split.append(V_direction - V_split[0])
741                self.lecs[lec] = 0.0
742                    
743        # we need GV, VG for each split V term
744        GV_split = []
745        VG_split = []
746        GVG_split = []
747        Vi_split_T11 = [] # the on-shell Vi-split terms
748        Vi_split_T12 = [] # the on-shell Vi-split terms
749        Vi_split_T22 = [] # the on-shell Vi-split terms
750        for this_V in V_split:
751            GV = self.setup_GV_kernel(channel[0],this_V,ko,mu)
752            VG = self.setup_VG_kernel(channel[0],this_V,ko,mu)
753            Vi_split_T11.append(this_V[Np,Np])
754            if (nof_blocks == 4):
755                Vi_split_T12.append(this_V[2*Np+1,Np])
756                Vi_split_T22.append(this_V[2*Np+1,2*Np+1])
757            GV_split.append(GV)
758            VG_split.append(VG)
759            GVG_split.append(GV@G)
760                
761        # loop over the training lec values and setup the relevant lec vector
762        # to assemble the potential for the split terms
763        Ti = []
764        for training_point in training_points:
765            lec_training_vector = []
766            lec_training_vector.append(1.0)
767            for lec, value in self.lecs.items():
768                if lec in directions:
769                    # training points and directions are identically ordered wrt lec names
770                    lec_training_vector.append(training_point[directions.index(lec)])
771
772            V = self.get_Vmtx_from_split(V_split,lec_training_vector)
773            this_T = self.solve_lippmann_schwinger(channel[0],V,ko,mu)
774            Ti.append(self.solve_lippmann_schwinger(channel[0],V,ko,mu))
775
776        # construct the emulator
777        mi_split_T11  = np.zeros((len(training_points)), dtype=object)
778        Mij_split_T11 = np.zeros((len(training_points),len(training_points)), dtype=object)
779        Mij_const_T11 = np.zeros((len(training_points),len(training_points)), dtype=object)
780
781        if (nof_blocks == 4):
782            mi_split_T12  = np.zeros((len(training_points)), dtype=object)
783            mi_split_T22  = np.zeros((len(training_points)), dtype=object)
784            Mij_split_T12 = np.zeros((len(training_points),len(training_points)), dtype=object)
785            Mij_split_T22 = np.zeros((len(training_points),len(training_points)), dtype=object)
786            Mij_const_T12 = np.zeros((len(training_points),len(training_points)), dtype=object)
787            Mij_const_T22 = np.zeros((len(training_points),len(training_points)), dtype=object)
788            
789        for r,training_point in enumerate(training_points):
790            mi_split_T11[r] = []
791            if (nof_blocks == 4):
792                mi_split_T12[r] = []
793                mi_split_T22[r] = []
794            for c,training_point in enumerate(training_points):
795                Mij_split_T11[r,c] = []
796                Mij_const_T11[r,c] = []
797                if (nof_blocks == 4):
798                    Mij_split_T12[r,c] = []
799                    Mij_const_T12[r,c] = []
800                    Mij_split_T22[r,c] = []
801                    Mij_const_T22[r,c] = []
802                    
803        for split_idx, _ in enumerate(GV_split):
804            for r,training_point in enumerate(training_points):
805                this_mi_TGV = np.copy(Ti[r]@GV_split[split_idx])
806                this_mi_VGT = np.copy(VG_split[split_idx]@Ti[r])
807                mi_split_T11[r].append((this_mi_TGV + this_mi_VGT)[Np,Np])
808                if (nof_blocks == 4):

103        The method uses the legP function to compute the Legendre polynomials at the quadrature points defined by self.z, which are the roots of the Legendre polynomial of degree self.ntheta. 104        """

106

108            this_P = fP(0,j,self.z)

159        """
160        This method performs the partial-wave decomposition of the central force component of the nucleon-nucleon potential. It takes as input the potential values W evaluated at the quadrature points for cos(theta), the momenta pp and p of the bra and ket states in the partial-wave decomposition, a flag coup indicating whether the channel is coupled or uncoupled, the total spin s of the two-nucleon system, and the total angular momentum j. The method calculates the central force components based on these inputs and returns an array V containing the decomposed components.
161
162        Args:
163            W (array-like): array of potential values evaluated at the quadrature points for cos(theta)
164            pp (float): momentum of the bra state in the partial-wave decomposition
165            p (float): momentum of the ket state in the partial-wave decomposition
166            coup (bool): flag indicating whether the channel is coupled (True) or uncoupled (False)
167            s (int): total spin of the two-nucleon system (0 for singlet, 1 for triplet)
168            j (int): total angular momentum of the two-nucleon system
169
170
171        Returns:
172            np.ndarray: array of partial-wave decomposed central force components
173        """
174        V = np.zeros(6)
175        
176        # uncoupled singlet
177        if not coup and s==0:
178            V[0] = 2*self.pwd_integral(W,0,j)
179        # uncoupled triplet
180        elif not coup and s==1:
181            V[1] = 2*self.pwd_integral(W,0,j)
182        elif coup:
183            V[2] = 2*self.pwd_integral(W,0,j+1)
184
185            # 3P0 case 
186            if j==0:
187                return V
188
189            V[3] = 2*self.pwd_integral(W,0,j-1)
190            V[4] = 0
191            V[5] = 0
192            
193        return V

234        """This method performs the partial-wave decomposition of the tensor force component of the nucleon-nucleon potential. The method calculates the tensor force components based on these inputs and returns an array V containing the decomposed components.
235
236        Args:
237            W (array-like): array of potential values evaluated at the quadrature points for cos(theta)
238            pp (float): momentum of the bra state in the partial-wave decomposition
239            p (float): momentum of the ket state in the partial-wave decomposition
240            coup (bool): flag indicating whether the channel is coupled (True) or uncoupled (False)
241            s (int): total spin of the two-nucleon system (0 for singlet, 1 for triplet)
242            j (int): total angular momentum of the two-nucleon system
243
244        Returns:
245            np.ndarray: array of partial-wave decomposed tensor force components
246        """
247        V = np.zeros(6)
248        
249        jj  = 2*j+1
250        jj1 = j*(j+1) 
251
252        pp2 = pp**2
253        p2  = p**2
254
255        # uncoupled singlet
256        if not coup and s==0:
257            V[0] = 2*( -1*(pp2+p2)*self.pwd_integral(W,0,j) + 2*pp*p*self.pwd_integral(W,1,j))
258        # uncoupled triplet
259        elif not coup and s==1:
260            V[1] = 2*( (pp2+p2)*self.pwd_integral(W,0,j) - 2*pp*p*(j*self.pwd_integral(W,0,j+1) + (j+1)*self.pwd_integral(W,0,j-1))/jj)
261        elif coup:
262            V[2] = 2*( -1*(pp2+p2)*self.pwd_integral(W,0,j+1) + 2*pp*p*self.pwd_integral(W,0,j))/jj
263
264            # 3P0 case 
265            if j==0:
266                return V
267
268            V[3] = 2*( (pp2+p2)*self.pwd_integral(W,0,j-1) - 2*pp*p*self.pwd_integral(W,0,j))/jj
269            V[4] = -4*np.sqrt(jj1)*(p*p*self.pwd_integral(W,0,j+1) + pp*pp*self.pwd_integral(W,0,j-1) - 2*pp*p*self.pwd_integral(W,0,j))/jj
270            V[5] = -4*np.sqrt(jj1)*(p*p*self.pwd_integral(W,0,j-1) + pp*pp*self.pwd_integral(W,0,j+1) - 2*pp*p*self.pwd_integral(W,0,j))/jj
271            
272        return V

421            pp (float): momentum of the bra state in the partial-wave decomposition
422            p (float): momentum of the ket state in the partial-wave decomposition
423            coup (bool): flag indicating whether the channel is coupled (True) or uncoupled (False)
424            s (int): total spin of the two-nucleon system (0 for singlet, 1 for triplet)
425            j (int): total angular momentum of the two-nucleon system

427            tz (int): projection of the total isospin along the z-axis

429
430        Returns:
431            np.ndarray: array containing the values of the nucleon-nucleon potential components in momentum space, calculated by summing contributions from one-pion exchange, two-pion exchange, and contact interactions based on the specified chiral order (LO or NLO). The potential is returned as an array of values corresponding to the different components of the potential in the partial-wave decomposition.
432        """
433
434        if coup:
435            assert s == 1, 'coupled NN channel must be a spin triplet'
436       
437        V = np.zeros(6)
438        
439        q2 = pp**2 + p**2 - 2*pp*p*self.z
440        if self.chiral_order == 'LO':
441        
442            WT = self.iso(t) * self.one_pion_exchange(q2)
443            V  += self.pwd_T(WT,pp,p,coup,s,j)
444            V  += self.contact_lo(pp,p,coup,s,j,t,tz,lecs)
445            
446        elif self.chiral_order == 'NLO':

449            # and use the isospin symmetric one-pion exhange
450            WT = self.iso(t) * self.one_pion_exchange(q2)
451            V  += self.pwd_T(WT,pp,p,coup,s,j)
452
453            V  += self.contact_lo(pp,p,coup,s,j,t,tz,lecs)
454            
455            # two-pion exchange loop function in dimensional regularization
456            loop = self.leading_two_pion_exchange_loop_DR(q2)
457
458            WC = self.iso(t) * self.leading_two_pion_exchange_WC(q2,loop)
459            V += self.pwd_C(WC,pp,p,coup,s,j)
460
461            VS = self.leading_two_pion_exchange_VS(q2,loop)
462            V += self.pwd_S(VS,pp,p,coup,s,j)
463
464            VT = self.leading_two_pion_exchange_VT(q2,loop)
465            V += self.pwd_T(VT,pp,p,coup,s,j)
466
467            V += self.contact_nlo(pp,p,coup,s,j,t,tz,lecs)
468                        
469        else:
470            print(f'only implemented LO and NLO in Weinberg counting.')        
471
472        # 1/2pi^3 normalization
473        return (0.125/np.pi**3)*self.freg(pp)*V*self.freg(p)
